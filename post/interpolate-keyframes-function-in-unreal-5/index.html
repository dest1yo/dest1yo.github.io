// FTransform Lerp
static FTransform LerpSomething(const FTransform& A, const FTransform& B, const float& Alpha)
{
	return FTransform {
			FQuat::Slerp(A.GetRotation(), B.GetRotation(), Alpha)
		, FMath::Lerp(A.GetLocation(), B.GetLocation(), Alpha)
		, FMath::Lerp(A.GetScale3D(), B.GetScale3D(), Alpha)
	};
}

// FVector Lerp

}

// FQuat SLerp

	return FQuat::Slerp(A, B, Alpha);
}
